24  void Read_JSON_FILE(std::vector<CONVERTER::IC*> *IC_INPUT,std::string folder_path,std::string FILE_NAME){
25  struct _finddata_t fd;
26  intptr_t handle;
27  string FIND_PATH = folder_path+"\\*.geojson";
28  if ((handle = _findfirst(FIND_PATH.c_str(), &fd)) == -1L)
29  cout << "No IMDF"+FILE_NAME+".geojson in directory!" << endl;
30  do{
31  if(strcmp(fd.name,(FILE_NAME+".geojson").c_str())==0){
32  FIND_PATH=folder_path+"\\"+FILE_NAME+".geojson";
33  if(FILE_NAME=="unit")
34  Read_UNIT(IC_INPUT, FIND_PATH);
35  if(FILE_NAME=="opening")
36  Read_OPENING(IC_INPUT,FIND_PATH);
37  if(FILE_NAME=="level")
38  Read_LEVEL(IC_INPUT,FIND_PATH);
39  }
40  } while (_findnext(handle, &fd) == 0);
41  _findclose(handle);
42  }
